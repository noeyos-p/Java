select * from city where city_name like %서울%;
where rigion in('서울', '부산') : 지역이 서울 또는 부산
where value between 10 and 20; : 10보다 크거나 같고, 20보다 작거나 같은
order by name asc : 오름차순
order by rigion asc, name desc : rigion 오름차순, desc 내림차순
